### 4A. How to Change Colors

Open `theme.js` and find the active theme (`THEMES.dark` by default). Edit the `colors` object:

```js
// frontend/src/theme.js

export const THEMES = {
  dark: {
    colors: {
      //
      // ── PRIMARY BRAND COLOR ──────────────────────────────────────────────
      // This is the main blue used on buttons, links, score rings, active states
      primary:       "#1A6BFF",   // ← Change this to your brand color
      primaryDark:   "#0047CC",   // ← Darker shade (hover/pressed)
      primaryGlow:   "rgba(26, 107, 255, 0.18)",  // ← Semi-transparent glow

      // ── ALERT / WARNING COLOR ────────────────────────────────────────────
      // Used on: missing skills, warnings, errors
      alert:         "#FF6425",   // ← Change to red, amber, etc.

      // ── SUCCESS / POSITIVE COLOR ─────────────────────────────────────────
      // Used on: owned skills, green checkmarks, growth arrows
      success:       "#AAFF00",   // ← Change to green, teal, etc.

      // ── BACKGROUND COLORS ────────────────────────────────────────────────
      bg:            "#141420",   // ← Darkest — page background
      bgCard:        "#1C1C2A",   // ← Card/panel surface
      bgInput:       "#252535",   // ← Input fields, dropdowns
      bgDeep:        "#0D0D18",   // ← Sidebar, modal overlays

      // ── TEXT COLORS ──────────────────────────────────────────────────────
      text:          "#EEEEF8",   // ← Primary text
      textMuted:     "#7777AA",   // ← Labels, secondary info
      textDim:       "#44445A",   // ← Placeholder, disabled text

      // ── BORDER COLORS ────────────────────────────────────────────────────
      border:        "rgba(255, 255, 255, 0.07)",  // ← Subtle dividers
      borderActive:  "rgba(26, 107, 255, 0.35)",   // ← Focused/selected border
    },
```

**To switch to light mode**, change the last line:
```js
// At bottom of theme.js:
const theme = THEMES.dark;   // ← Change to: THEMES.light
export default theme;
```

### 5B. How to Replace Dummy Salary Data with Real Data

**Step 1 — Collect and clean the data:**

Create `backend/ml_training/collect_salary_data.py`:
```python
import pandas as pd
import requests

# Download Stack Overflow survey
df = pd.read_csv("survey_results_public.csv")

# Filter and clean
salary_df = df[["ConvertedCompYearly", "YearsCodePro", "DevType",
                "Country", "EdLevel", "LanguageHaveWorkedWith"]].dropna()

# Rename columns to match our schema
salary_df = salary_df.rename(columns={
    "ConvertedCompYearly": "salary_usd",
    "YearsCodePro":        "years_experience",
    "DevType":             "role",
    "Country":             "location",
    "EdLevel":             "education",
    "LanguageHaveWorkedWith": "skills_raw",
})

# Remove outliers (keep $20K–$500K)
salary_df = salary_df[
    (salary_df["salary_usd"] >= 20_000) &
    (salary_df["salary_usd"] <= 500_000)
]

salary_df.to_csv("clean_salary_data.csv", index=False)
print(f"Saved {len(salary_df):,} salary records")
```

**Step 2 — Train the salary model:**
```bash
cd backend
python ml_training/train_salary_model.py --data clean_salary_data.csv
```

The `train_salary_model.py` script:
- Loads the cleaned CSV
- Feature engineers: role embedding, location factor, education one-hot
- Trains: `GradientBoostingRegressor` (or PyTorch MLP)
- Saves model to: `backend/models/salary_model.pkl`
- Evaluates: MAE, RMSE, R² score

**Step 3 — Load model in ml_engine.py:**
```python
# In ml_engine.py — replace the deterministic function:
import joblib

_salary_model = None
def _load_salary_model():
    global _salary_model
    if _salary_model is None:
        try:
            _salary_model = joblib.load("models/salary_model.pkl")
        except FileNotFoundError:
            pass  # Fall back to deterministic algorithm
    return _salary_model

def salary_estimate_from_features(role, years_exp, skills, location, education="bachelor"):
    model = _load_salary_model()
    if model:
        # Build feature vector matching training schema
        features = encode_features(role, years_exp, skills, location, education)
        prediction = model.predict([features])[0]
        return build_response(prediction)
    else:
        return _deterministic_estimate(role, years_exp, skills, location, education)
```

```python
# job_matcher.py — fetch_live_jobs() method

async def fetch_live_jobs(self, role: str, location: str = "us") -> list:
    """Fetch real jobs from Adzuna API."""
    if not (self.config.adzuna_app_id and self.config.adzuna_api_key):
        return self._get_sample_jobs()  # fall back to demo data

    url = (
        f"https://api.adzuna.com/v1/api/jobs/{location}/search/1"
        f"?app_id={self.config.adzuna_app_id}"
        f"&app_key={self.config.adzuna_api_key}"
        f"&results_per_page=20"
        f"&what={role}"
        f"&content-type=application/json"
    )

    async with httpx.AsyncClient() as client:
        resp = await client.get(url, timeout=10)
        data = resp.json()

    jobs = []
    for item in data.get("results", []):
        jobs.append({
            "id":                  item["id"],
            "title":               item["title"],
            "company":             item["company"]["display_name"],
            "location":            item["location"]["display_name"],
            "salary_min":          item.get("salary_min", 0),
            "salary_max":          item.get("salary_max", 0),
            "description":         item.get("description", ""),
            "redirect_url":        item["redirect_url"],
            "required_skills":     self._extract_skills_from_description(item.get("description","")),
            "posted_date":         item["created"],
        })
    return jobs
```

### 5D. How to Train the Skill Similarity / NLP Model

**Current:** Keyword matching against a hardcoded skills taxonomy.

**Upgrade to:** Sentence-BERT embeddings for semantic skill matching.

```bash
# Install
pip install sentence-transformers

# Then in ml_training/train_job_matcher.py
```

```python
from sentence_transformers import SentenceTransformer, util
import torch

# Load pre-trained model (downloads ~90MB on first run)
model = SentenceTransformer("all-MiniLM-L6-v2")

def compute_semantic_skill_overlap(user_skills: list, job_skills: list) -> float:
    """Uses embeddings instead of exact keyword matching."""
    if not user_skills or not job_skills:
        return 0.0

    user_embs = model.encode(user_skills, convert_to_tensor=True)
    job_embs  = model.encode(job_skills,  convert_to_tensor=True)

    # Cosine similarity matrix
    sim_matrix = util.cos_sim(user_embs, job_embs)

    # Best match score for each job skill
    per_job_skill_score = sim_matrix.max(dim=0).values
    return float(per_job_skill_score.mean()) * 100
```

**Drop this into `ml_engine.py`** to replace the `compute_similarity_score` function's `SkillOverlap` calculation.
